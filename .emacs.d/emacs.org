#+begin_src elisp
;; Extension functions for exwm
(defun exwm-ext-close-all-windows ()
  "Closes all open windows in the frame, leaving just one open."
  (while (> (count-windows) 1)
    (evil-window-delete)))

(defun exwm-ext-open-in-splits (names)
  "Closes all open windows in the frame, and replaces them with them
  X11 programs provided, tiled horizontally.
  e.g. (exwm-ext-open-in-splits '(\"firefox\" \"gnome-terminal\"))
  will replace the current frame with two windows tiled one above the other,
  with firefox at the top and gnome-terminal at the bottom."
  (exwm-ext-close-all-windows)
  (let ((i (length names)))
    (while (> i 1)
      (evil-window-split)
      (sleep-for 1)
      (setq i (1- i))))
  (message "split complete")
  (sleep-for 5)
  (let (name) 
    (dolist (name names)
      (message name)
      (sleep-for 1)
      (start-process-shell-command name nil name)
      (sleep-for 1)
      (other-window 1))))

;; Function for synchronising notes files with git
(defun sync-notes()
  (when (string-match-p (regexp-quote "notes") buffer-file-name)
    (magit-stage-file buffer-file-name)
    (magit-commit-create (list "-m" "auto-commit from emacs"))
    (magit-fetch-all ())
    (magit-rebase-branch "origin/master" ())
    (magit-push-current-to-upstream ())))

** Experimental go log highlighting
This function (~go-log-highlight~) is a prototype for a minor mode that could be used
to highlight log output from ~go test~. The idea is that you would open a buffer containing
log output from ~go test~, switch to ~Fundamental mode~, and then run ~go-log-highlight~.
#+begin_src elisp
(defgroup go-log-highlight-group nil
  "Go log highlighting"
  :group 'go)

(defface go-log--fail-face
  '((t (:foreground "red")))
  "Fail face"
  :group 'go-log-highlight-group)

(defface go-log--run-face
  '((t (:foreground "magenta")))
  "Run face"
  :group 'go-log-highlight-group)

(defface go-log--pass-face
  '((t (:foreground "green")))
  "Pass face"
  :group 'go-log-highlight-group)

(defface go-log--info-face
  '((t (:foreground "green")))
  "Info face"
  :group 'go-log-highlight-group)

(defface go-log--warning-face
  '((t (:foreground "yellow")))
  "Warning face"
  :group 'go-log-highlight-group)

(defface go-log--error-face
  '((t (:foreground "red")))
  "Error face"
  :group 'go-log-highlight-group)

(defface go-log--field-face
  '((t (:foreground "gray")))
  "Field face"
  :group 'go-log-highlight-group)

(defface go-log--string-face
  '((t (:foreground "cyan")))
  "Field face"
  :group 'go-log-highlight-group)

(defface go-log--uuid-face
  '((t (:foreground "salmon")))
  "Field face"
  :group 'go-log-highlight-group)

(defface go-log--number-face
  '((t (:foreground "khaki")))
  "Field face"
  :group 'go-log-highlight-group)

(defvar go-log-highlight-keywords
  '(
    ("^\s*=== RUN" . 'go-log--run-face t)
    ("^\s*--- PASS" . 'go-log--pass-face t)
    ("^\s*--- FAIL" . 'go-log--fail-face t)
    ("level=\\(info\\)" 1 'go-log--info-face t)
    ("level=\\(warning\\)" 1 'go-log--warning-face t)
    ("level=\\(error\\)" 1 'go-log--warning-face t)
    ("[a-zA-z\\-]+=" . 'go-log--field-face t)
    ("=\\(\"[^\"]+\\)" 1 go-log--string-face t)
  ))

(define-minor-mode go-log-highlight-minor-mode
  "Minor mode for making go logs more readable"
  :init-value 1
  :group 'go-log-highlight-group
  (when (bound-and-true-p go-log-highlight-minor-mode)
    (font-lock-add-keywords nil go-log-highlight-keywords)
    (font-lock-fontify-buffer))
  (when (not (bound-and-true-p go-log-highlight-minor-mode))
    (font-lock-remove-keywords nil go-log-highlight-keywords)
    (font-lock-fontify-buffer)))
#+end_src
